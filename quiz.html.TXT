<script>

let sorular = [
    { dogru: "birçok", yanlis: "bir çok" },
    { dogru: "başkent", yanlis: "baş kent" },
    { dogru: "hiçbir", yanlis: "hiç bir" },
    { dogru: "art arda", yanlis: "artarda" },
    { dogru: "her şey", yanlis: "herşey" }
];

let puan = 0;
let aktifSoru;

function yeniSoru() {

    let rastgele = Math.floor(Math.random() * sorular.length);
    aktifSoru = sorular[rastgele];

    let secenekler = [aktifSoru.dogru, aktifSoru.yanlis];

    // Karıştır
    secenekler.sort(() => Math.random() - 0.5);

    let s1 = document.getElementById("secenek1");
    let s2 = document.getElementById("secenek2");

    s1.className = "option";
    s2.className = "option";

    s1.innerText = secenekler[0];
    s2.innerText = secenekler[1];

    s1.onclick = () => kontrol(secenekler[0], s1, s2);
    s2.onclick = () => kontrol(secenekler[1], s2, s1);
}

function kontrol(secim, tiklanan, diger) {

    if(secim === aktifSoru.dogru) {
        puan++;
        document.getElementById("puan").innerText = puan;
        tiklanan.classList.add("correct");
    } else {
        tiklanan.classList.add("wrong");
        diger.classList.add("correct");
    }

    setTimeout(yeniSoru, 1200);
}

yeniSoru();
